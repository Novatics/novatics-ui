### Installation

First install the latest version of the package from `@novatics/backdrop`.

```shell
  yarn add -D @novatics/backdrop
```

```shell
  npm install --dev @novatics/backdrop
```

and use it

```tsx
import ... from '@novatics/backdrop';

  ...


```

### Basic Usage

<Canvas>
  <Story name="Basic Usage">
    {(args) => {
      const [openModal, setOpenModal] = useState(false);
      const [openLoading, setOpenLoading] = useState(false);
  
      return (
        <>
          <Stack spacing={2} direction={"row"}>
            <Button onClick={()=>setOpenModal(true)}>Show backdrop with a modal</Button>
            <Button onClick={()=>setOpenLoading(true)}>Show backdrop with a loading icon</Button>
          </Stack>

          <Backdrop open={openModal} onClick={()=>setOpenModal(false)}>
            <div style={{background: "white", color: "black", padding: '40px'}}>
              <h1>Modal</h1>
            </div>
          </Backdrop>
          
          <Backdrop open={openLoading} onClick={()=>setOpenLoading(false)}>
            <CircularProgress />
          </Backdrop>
        </>
      );
    }}

  </Story>
</Canvas>
